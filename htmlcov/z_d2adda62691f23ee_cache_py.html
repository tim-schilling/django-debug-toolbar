144 def _record_call(self, cache, alias, name, original_method, args, kwargs): 

145 # Some cache backends implement certain cache methods in terms of other cache 

146 # methods (e.g. get_or_set() in terms of get() and add()). In order to only 

147 # record the calls made directly by the user code, set the cache's _djdt_panel 

148 # attribute to None before invoking the original method, which will cause the 

149 # monkey-patched cache methods to skip recording additional calls made during 

150 # the course of this call, and then reset it back afterward. 

151 cache._djdt_panel = None

194 def enable_instrumentation(self): 

195 # Monkey patch all open cache connections. Django maintains cache connections 

196 # on a per-thread/async task basis, so this will not affect any concurrent 

197 # requests. The monkey patch of CacheHander.create_connection() installed in 

198 # the .ready() method will ensure that any new cache connections that get opened 

199 # during this request will also be monkey patched. 

200 for cache, alias in self.initialized_caches():

202 # Mark this panel instance as the current one for the active thread/async task 

203 # context. This will be used by the CacheHander.create_connection() monkey 

204 # patch.
